export async function runAgent(userInput, messages, depth = 0) {
  messages.push({ role: "user", content: userInput });
  while (true) {
    await compactIfNeeded(messages);          // ⑥ 太长先压缩
    const res = await withRetry(async () => {  // ⑩ 失败自动重试
      const stream = client.messages.stream({
        model: MODEL, system: buildSystemPrompt(),  // ④人设+⑦记忆
        messages, tools: toolSchemas,
      });
      stream.on("text", (c) => process.stdout.write(c)); // ⑤流式
      return await stream.finalMessage();
    });
    messages.push({ role: "assistant", content: res.content });
    if (res.stop_reason !== "tool_use") return finalText;  // 没工具→完成
    // …执行工具(②③⑧⑨⑫)，把结果喂回 messages，进下一轮…
  }
}

src/permission.js · checkPermission()
export async function checkPermission(tool, input) {
  const action = matchRule(tool.name, input);   // ⑨ 先查规则表
  if (action === "allow") return true;       // 自动放行
  if (action === "deny")  return false;      // 直接拒
  if (action === null && !tool.needsApproval) return true;
  if (sessionAllowlist.has(tool.name)) return true; // 记住过
  const ans = await ask(`允许吗? (y / n / a=以后都允许) `);
  const c = ans.trim().toLowerCase();
  if (c === "a") { sessionAllowlist.add(tool.name); return true; }
  return c === "y";
}

export async function compactIfNeeded(messages) {
  if (messages.length <= MAX_HISTORY) return;  // 没超就不动
  let keepStart = Math.max(0, messages.length - KEEP_RECENT);
  const oldMessages = messages.slice(0, keepStart);
  const recentMessages = messages.slice(keepStart);
  // …把 oldMessages 转成文本，调模型总结成 summary…
  const summaryMsg = { role: "user",
    content: `【以上对话摘要】\n${summary}` };
  messages.splice(0, messages.length,        // 原地替换
                  summaryMsg, ...recentMessages);
  console.log(`📦 已把前面 ${oldMessages.length} 条压成摘要`);
}

const dangerousCmdHook = (toolName, input) => {
  if (toolName !== "run_bash") return null;
  const cmd = String(input.command || "");
  const dangerous = [
    /rm\s+-rf?\s+\/[^a-z]/i,   // rm -rf /
    /rm\s+-rf?\s+\./i,         // rm -rf .
    /mkfs\s*\./i, /dd\s+if=/i,    // 格式化 / 覆写磁盘
  ];
  for (const re of dangerous)
    if (re.test(cmd)) return { block: true, reason: "毁灭性命令" };
  return null;   // 放行（再交给护栏问人）
};

src/permission.js
src/permission.js · permissionRules
export const permissionRules = [
  { tool: "run_bash",
    match: (i) => /^\s*git\s+(status|diff|log)\b/.test(i.command),
    action: "allow" },          // 只读 git 自动放行
  { tool: "run_bash",
    match: (i) => /\bsudo\b/.test(i.command),
    action: "deny" },           // sudo 直接拒，连问都不问
];
export function matchRule(toolName, input) {
  for (const rule of permissionRules) {     // 从头扫，第一条命中即返回
    if (rule.tool !== toolName) continue;
    if (rule.match && !rule.match(input)) continue;
    return rule.action;
  }
  return null;
}

容错重试

进阶 · 更稳更强 · withRetry()
🎯
API 抖一下别跟着崩
指数退避重试
网络会抖、会限流。包一层重试：失败按 0.5s→1s→2s 翻倍等待(+随机抖动)，4xx 立刻抛不浪费。
  • 429/超时可重试，401 鉴权错立刻抛
  • 对照 services/api/withRetry.ts
src/retry.js
src/retry.js · withRetry()
export async function withRetry(fn, opts = {}) {
  const { retries = 3, baseDelay = 500,
          isRetriable = isRetriableError } = opts;
  for (let attempt = 0; attempt <= retries; attempt++) {
    try { return await fn(); }
    catch (err) {
      if (attempt >= retries || !isRetriable(err)) throw err;
      const delay = baseDelay * Math.pow(2, attempt)
                  + Math.floor(Math.random() * 100);  // 指数退避+抖动
      await new Promise((r) => setTimeout(r, delay));
    }
  }
}

export function saveSession(messages) {
  const jsonl = messages.map((m) => JSON.stringify(m)).join("\n");
  writeFileSync(SESSION_FILE, jsonl, "utf8");     // 每条一行
}
export function loadSession() {
  if (!existsSync(SESSION_FILE)) return [];
  const raw = readFileSync(SESSION_FILE, "utf8").trim();
  return raw.split("\n").reduce((acc, line) => {
    try { acc.push(JSON.parse(line)); }   // 坏行跳过，不崩
    catch {}
    return acc;
  }, []);
}
